\oppgave{1}
\oppgaveDelStart

\begin{align*}
&A = \{x \in \mathbb{Z} | (2x - 1)(2x + 1) = 0\}\\
&B = \{x \in \mathbb{R} | (2x - 1)(2x + 1) = 0\}\\
&\text{Given\ } (2x-1)(2x+1) = 0 \Leftrightarrow 2x - 1 = 0 \Rightarrow x = \frac{1}{2} \text{\ or\ } 2x + 1 = 0 \Rightarrow x = -\frac{1}{2}\\
\end{align*}

\oppgaveDel{a}
\begin{align*}
&A = \emptyset\\
&\text{Since $x = \frac{1}{2}$ and $-\frac{1}{2}$ are not integers}
\end{align*}

\oppgaveDel{b}
\begin{align*}
&B = \{-\frac{1}{2},\frac{1}{2}\}
\end{align*}
\oppgaveDelSlutt

pq_tabell = [[True,True],[True,False],[False,True],[False,False]]
